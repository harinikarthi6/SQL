 CREATE TABLE PRODUCTLINES
 (
 PRODUCTLINE VARCHAR(100) PRIMARY KEY,
 TEXTDESCRIPTION VARCHAR(100),
 HTMLDESCRIPTION CLOB,
 IMAGE BLOB
 );

 CREATE TABLE PRODUCTS
 (
 PRODUCTCODE  VARCHAR(100) PRIMARY KEY,
 PRODUCTNAME VARCHAR(50) NOT NULL,
 PRODUCTLINE VARCHAR(100),
 PRODUCTSCALE VARCHAR(100) NOT NULL,
 PRODUCTVENDOR VARCHAR(100),
 PRODUCTDESCRIPTION CLOB,
 QUANTITYINSTOCK NUMBER(15),
 BUYPRICE NUMBER(12,4) NOT NULL,
 MSRP NUMBER(12,4) NOT NULL,
 FOREIGN KEY (PRODUCTLINE) REFERENCES PRODUCTLINES(PRODUCTLINE)
 );

CREATE TABLE OFFICES
(
OFFICECODE VARCHAR(100) PRIMARY KEY,
CITY VARCHAR(50),
PHONE NUMBER(10) CHECK(LENGTH(PHONE)=10),
ADDRESSLINE1 VARCHAR(100),
ADDRESSLINE2 VARCHAR(100),
STATE VARCHAR(50),
COUNTRY VARCHAR(50),
POSTALCODE  VARCHAR(100),
TERRITORY VARCHAR(100)
);

CREATE TABLE EMPLOYEES
(
EMPLOYEENUMBER NUMBER(10) PRIMARY KEY,
LASTNAME VARCHAR(100),
FIRSTNAME VARCHAR(100),
EXTENSION VARCHAR(200),
EMAIL VARCHAR(50) UNIQUE NOT NULL,
OFFICECODE VARCHAR(100),
REPORTSTO NUMBER(30),
JOBTITLE VARCHAR(50),
FOREIGN KEY (OFFICECODE) REFERENCES OFFICES(OFFICECODE),
FOREIGN KEY (REPORTSTO) REFERENCES EMPLOYEES(EMPLOYEENUMBER)
);

CREATE TABLE CUSTOMERS
(
CUSTOMERNUMBER NUMBER(10) PRIMARY KEY,
CUSTOMERNAME VARCHAR(50) NOT NULL,
CONTACTLASTNAME VARCHAR(50) NOT NULL,
CONTACTFIRSTNAME VARCHAR(50) NOT NULL,
PHONE NUMBER(10),
ADDRESSLINE1 VARCHAR(100),
ADDRESSLINE2 VARCHAR(100),
CITY VARCHAR(20),
STATE VARCHAR(20),
POSTALCODE VARCHAR(30),
COUNTRY VARCHAR(10),
SALESREPEMPLOYEENUMBER NUMBER(10),
CREDITLIMIT NUMBER(10,2),
FOREIGN KEY (SALESREPEMPLOYEENUMBER) REFERENCES EMPLOYEES(EMPLOYEENUMBER)
);

CREATE TABLE ORDERS
(
ORDERNUMBER NUMBER(10) PRIMARY KEY,
ORDERDATE DATE NOT NULL,
REQUIREDDATE DATE,
SHIPPEDDATE DATE,
STATUS VARCHAR(50),
COMMENTS VARCHAR(100),
CUSTOMERNUMBER NUMBER(10),
FOREIGN KEY (CUSTOMERNUMBER) REFERENCES CUSTOMERS(CUSTOMERNUMBER)
);


CREATE TABLE ORDERDETAILS
(
ORDERNUMBER NUMBER(10),
PRODUCTCODE VARCHAR(50),
QUANTITYORDERED NUMBER(10),
PRICEEACH NUMBER(10,2),
ORDERLINENUMBER NUMBER(20),
PRIMARY KEY (ORDERNUMBER,PRODUCTCODE),
FOREIGN KEY (ORDERNUMBER) REFERENCES ORDERS(ORDERNUMBER)
);  

CREATE TABLE PAYMENTS
(
CUSTOMERNUMBER NUMBER(20),
CHECKNUMBER NUMBER(20),
PAYMENTDATE DATE,
AMOUNT NUMBER(10,2),
PRIMARY KEY (CUSTOMERNUMBER, CHECKNUMBER),
FOREIGN KEY (CUSTOMERNUMBER) REFERENCES CUSTOMERS(CUSTOMERNUMBER)
);
